🔗 เชื่อมต่อ Google Sheet
สถานะ: ยังไม่ได้ตั้งค่า
📖 วิธีตั้งค่า Apps Script (v2 — รองรับ sync)
1
สร้าง Google Sheet ใหม่
ไปที่ sheets.google.com → สร้าง Spreadsheet ใหม่ → ตั้งชื่อ Sheet แรกว่า Data
2
เปิด Apps Script
เมนูบน Extensions → Apps Script
3
วาง Code ด้านล่าง (เวอร์ชัน sync)
ลบ code เดิมออกทั้งหมด แล้ววาง code นี้แทน:
const SHEET_NAME = "Data"; function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName(SHEET_NAME); if (sheet.getLastRow() === 0) { var h = sheet.getRange(1,1,1,6); sheet.appendRow(["วันที่","เวลา","ประเภท","รายละเอียด","จำนวนเงิน","ช่องทาง"]); h.setFontWeight("bold") .setBackground("#2e1f16") .setFontColor("#fdf8f2"); } var now = new Date(); sheet.appendRow([ Utilities.formatDate(now,"Asia/Bangkok","dd/MM/yyyy"), Utilities.formatDate(now,"Asia/Bangkok","HH:mm"), data.type, data.detail, parseFloat(data.amount), data.channel ]); return json({status:"ok"}); } catch(err) { return json({status:"error",msg:err.toString()}); } } function doGet(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName(SHEET_NAME); var rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return json({status:"ok",data:[]}); var records = rows.slice(1).map(function(r){ return { date: r[0], time: r[1], type: r[2], detail: r[3], amount: r[4], channel: r[5] }; }).reverse(); return json({status:"ok",data:records}); } catch(err) { return json({status:"error",msg:err.toString()}); } } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
4
Deploy เป็น Web App
Deploy → New deployment
Type: Web app  |  Execute as: Me
Who has access: Anyone → กด Deploy
คัดลอก URL → วางในช่องด้านบน
5
แชร์แอพให้แฟน
ส่งไฟล์ .html และ URL เดิมให้แฟน → ให้กรอก URL เดียวกันในแท็บตั้งค่า → ข้อมูลจะรวมใน Sheet เดียวกันและ sync ข้ามเครื่องได้เลย 🎉
