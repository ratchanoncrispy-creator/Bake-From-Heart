const SHEET_NAME = "Data"; function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName(SHEET_NAME); if (sheet.getLastRow() === 0) { sheet.appendRow(["วันที่","เวลา","ประเภท","สินค้า","รายละเอียด","จำนวน","ยอดขาย","% GP","ค่า GP","ส่วนลดแคมเปญ","ยอดสุทธิ","ช่องทาง"]); sheet.getRange(1,1,1,12).setFontWeight("bold") .setBackground("#2e1f16").setFontColor("#fdf8f2"); } sheet.appendRow([ data.date, data.time, data.type, data.product || "", data.detail || "", data.qty ? parseFloat(data.qty) : 1, parseFloat(data.amount), data.gp_pct !== "" ? parseFloat(data.gp_pct) : "", data.gp_amt !== "" ? parseFloat(data.gp_amt) : "", data.discount !== "" ? parseFloat(data.discount) : "", parseFloat(data.net_amount), data.channel || "" ]); return buildResponse({status:"ok"}); } catch(err) { return buildResponse({status:"error",msg:err.toString()}); } } function doGet(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName(SHEET_NAME); var rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return buildResponse({status:"ok",data:[]}); var records = rows.slice(1).map(function(r){ return { date: r[0], time: r[1], type: r[2], product: r[3], detail: r[4], qty: r[5], amount: r[6], gp_pct: r[7], gp_amt: r[8], discount: r[9], net_amount: r[10], channel: r[11] }; }).reverse(); return buildResponse({status:"ok", data:records}); } catch(err) { return buildResponse({status:"error",msg:err.toString()}); } } function buildResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
